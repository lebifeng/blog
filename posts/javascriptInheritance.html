# ES6的class继承

class Parent {
  x = 1

  constructor(name, age) {
    this.name = name
    this.age = age
    this.x = 2
  }

  static weight = '100KG'

  sayHello() {
    return `hello, this is ${this.name}`
  }

  static getWeight() {
    return this.weight
  }

  getAge() {
    return this.age
  }

}

Parent.prototype.y = 1

class Child extends Parent {

  constructor(name, age, school) {
    super(name, age)
    this.school = school
  }

  static weight = '30KG'

  normalMethodTest() {

    // 这里的super对象指向父类原型对象
    // super.y === Parent.prototype.y

    // 注意, ES6规定, 此处相当于super.sayHello.call(this)
    return super.getAge()
  }

  static staticMethodTest() {

    // 静态方法内的super指向父类, 且调用的静态方法内的this指向子类
    return super.getWeight()
  }
}

let parent = new Parent('father', 34)
let child = new Child('child', 14, 'HFUT')
child.sayHello()
parent.sayHello()

child.normalMethodTest()
Child.staticMethodTest()

// 输出
hello, this is child
hello, this is father
14  // 子类普通方法通过super调用父类原型上的方法, 方法内this指向子类实例
30KG // 子类静态方法通过super调用父类静态方法, 方法内this指向子类

# ES5的继承

1.构造函数继承


function Father(name, age) {
  this.name = name
  this.age = age
}
// 父类原型对象方法
Father.prototype.sayHi = function () {
  return this.age
}

function Son() {
  Father.call(this, ...arguments)
}

let son = new Son('jack', 25)
let father = new Father('father', 44)
console.log(son) // { name: 'jack', age: 25 }
console.log(father) // { name: 'father', age: 44 }
father.sayHi() // 44
son.sayHi() // 报错, 说明不能调用父类原型对象方法

// 缺点， 只能继承构造器属性, 无法共享原型方法

2.原型链继承

function Father(name) {
  this.name = name
}

Father.prototype.sayHi = function () {
  return `hi, this is ${this.name}`
}

Father.prototype.getCars = function () {
  return this.cars
}

Father.prototype.cars = ['bmw', 'benz']

function Son(name) {
  this.name = name
}

// 子类原型对象指向父类实例, 为什么要用实例, 而不是指向父类原型?因为父类也可能有父类, 用原型的话无法访问祖父类的原型对象
Son.prototype = new Father('father')
// 修改构造函数的原型会默认修改原型对象的构造器指向, 此处要改回子类
Son.prototype.constructor = Son

let son1 = new Son('jack')
let son2 = new Son('rose')
son1.sayHi() // hi, this is jack
son2.sayHi() // hi, this is rose
son1.getCars() // [ 'bmw', 'benz' ]
son1.cars.push('audi') 
console.log(son2.getCars()) // [ 'bmw', 'benz', 'audi' ]

// 可以看到, 引用类型的数据受其他子类影响,存在一定弊端

3.组合继承(构造函数 + 原型链)

function Father(name, age) {
  this.name = name
  this.age = age
  // 引用类型的实例属性
  this.names = []
}

Father.prototype.getNames = function() {
  return this.names
}

function Son(name, age, job) {
  // 借用构造函数
  // this执行的过程中也会创建this.names实例属性
  Father.call(this, name, age)
  this.job = job
}

// 创建原型链
// 需要注意此时Son类的原型对象中还是有Father类的实例对象的属性和方法
Son.prototype = new Father()

// 调整Son原型对象的constructor指向
Son.prototype.constructor = Son

let son1 = new Son('bob', 24, 'web')
let son2 = new Son('jack', 28, 'paper')
son1.names.push('bob Lee')
son2.names.push('jack chen')
son1 // Son { name: 'bob', age: 24, names: [ 'bob Lee' ], job: 'web' }
son2 // Son { name: 'jack', age: 28, names: [ 'jack chen' ], job: 'paper' }
