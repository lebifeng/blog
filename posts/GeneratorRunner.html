# 基于Promise的Generator函数自动执行器

# async函数 -- Generator的语法糖

了解 async函数 和 Generator函数的小伙伴一定会很好奇, 这俩怎么长得这么像啊?

其实async函数就是Generator函数的语法糖, 在Generator的基础上封装了一层自动执行函数而已

eg:


function* gen () {
    let a = yield new Promise()
    let b = yield new Promise()
    return 
}

async function asyncFn () {
    let res1 = await new Promise()
    let res2 = await new Promise()
    return res2
}
// ??双胞胎呢这是

// generator函数的执行
let g = gen()
let a = g.next()
let b = g.next()

// generator函数的执行依赖next方法的调用, 而async函数只需调用一次即可
// async 函数执行
let res = asyncFn().then(res => {})

# 那么怎么让一个Generator函数也自动执行呢 ?

function* gen() {

  let result1 = yield new Promise(function (resolve, reject) {
    setTimeout(function () {
      resolve(1)
    }, 2000)
  })

  let result2 = yield new Promise(function (resolve, reject) {
    resolve(1 + result1)
  })

  return result1 + result2
}

let g = gen()

function run(generator) {
  let g = generator()

  // 关键函数
  function next(data) {
    let result = g.next(data)
    console.log(result)
    if (result.done) {
      return result.value
    }
    // 如果没有done, 则返回的是一个promise对象, 调用then方法即可 
    result.value.then(function (data) {
      next(data)
    })
  }

  next()
}

run(gen)

// 输出
{ value: Promise { <pending> }, done: false }
{ value: Promise { 2 }, done: false }
{ value: 3, done: true }
