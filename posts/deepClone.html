# 浅克隆

let obj = {
  name: 'jack',
  age: Symbol('26'),
  male: true,
  location: {
    x: 12,
    y: 22
  },
  arr: [1, 2, 3],
  get: function () {
    console.log('origin get')
  }
}
obj.get.valueOf = function() {
  return 2
}

let newObj = {}

// let newObj = Object.assign({}, obj)
for (let key in obj) {
  if (obj.hasOwnProperty(key)) {
    newObj[key] = obj[key]
  }
}

console.log(newObj)

obj.male = false

3.通用方法

function deepClone(object) {
  let type = getRealType(object)
  let newObj
  // 基本类型 直接赋值
  if (simpleDataType(type)) {
    return object
  }

  // 对象和函数 均需要遍历属性, 根据属性值再判断是否需要继续深克隆
  if (type === 'Object' || type === 'Function') {
    if (type === 'Function') {
      newObj = function () {
        return object.call(this, ...arguments)
      }
    } else {
      newObj = {}
    }
    for (let key in object) {
      if (!object.hasOwnProperty(key)) {
        continue
      }
      let type = getRealType(object[key])
      if (simpleDataType(type)) {
        newObj[key] = object[key]
      } else {
        newObj[key] = deepClone(object[key])
      }
    }
    return newObj
  }

  // 数组的深克隆
  if (type === 'Array') {
    return [...object]
  }

}

// 判断是否基本类型
function simpleDataType(type) {
  return type === 'Number' || type === 'String' || type === 'Symbol' || type === 'Boolean' || type === 'Null'
}

// 获取对象的实际类型
function getRealType(object) {
  return Object.prototype.toString.call(object).slice(8, -1)
}
