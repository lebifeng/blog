# 手写promise

# 一、简单版本

const STATUS = {
  pending: 'pending',
  fulfilled: 'fulfilled',
  rejected: 'rejected'
}

/**
 * constructor和原生Promise相同function(resolve, reject) {}
 * @param constructor
 * @constructor
 */
function SimplePromise(constructor) {
  this.value = undefined
  this.reason = undefined
  this.onfulfilledCallbacks = []
  this.onrejectedCallbacks = []
  this.status = STATUS.pending

  this.resolve = (value) => {
    console.log('resolve was called')
    if (this.status === STATUS.pending) {
      this.status = STATUS.fulfilled
      console.log('resolved')
      this.value = value
      this.onfulfilledCallbacks.forEach(cb => {
        cb(this.value)
      })
    }
  }

  this.reject = (reason) => {
    console.log('reject was called')
    if (this.status === STATUS.pending) {
      this.status = STATUS.rejected
      console.log('rejected')
      this.reason = reason
      this.onrejectedCallbacks.forEach(cb => {
        cb(this.reason)
      })
    }
  }
  try {
    constructor(this.resolve, this.reject)
  } catch (e) {
    // 传的回调函数 不是 function(resolve, reject){}
    console.error(e)
  }
}

SimplePromise.prototype.then = function (onResolved, onRejected) {
  console.log('then was called')
  if (this.status === STATUS.pending) {
    console.log('still pending')
    this.onfulfilledCallbacks.push(onResolved)
    this.onrejectedCallbacks.push(onRejected)
  }

  if (this.status === STATUS.fulfilled) {
    onResolved(this.value)
  }

  if (this.status === STATUS.rejected) {
    onRejected(this.reason)
  }
}

优点:

实现简单， 支持异步

new SimplePromise(function (resolve, reject) {
  setTimeout(function () {
    resolve(123)
  }, 1000)
}).then(res => {
  console.log(res)
}, err => {
  console.error(err)
})

// 输出
then was called
still pending
resolve was called
resolved
我是异步调用的

缺点：

不支持链式调用. then回调之后不能再调用then，否则报错

# 二、进阶版本

const STATUS = {
  pending: Symbol(),
  fulfilled: Symbol(),
  rejected: Symbol()
}
const KEYS = {
  status: Symbol(),
  reason: Symbol(),
  value: Symbol(),
  fulfilledCallbacks: Symbol(),
  rejectedCallbacks: Symbol()
}

function MyPromise(constructor) {
  let self = this
  self[KEYS.status] = STATUS.pending
  self[KEYS.value] = undefined
  self[KEYS.reason] = undefined
  self[KEYS.fulfilledCallbacks] = []
  self[KEYS.rejectedCallbacks] = []

  function resolve(value) {
    console.log('resolve was called')
    if (self[KEYS.status] === STATUS.pending) {
      self[KEYS.status] = STATUS.fulfilled
      self[KEYS.value] = value
      self[KEYS.fulfilledCallbacks].forEach(callback => callback(value))
    }
  }

  function reject(reason) {
    console.log('reject was called')
    if (self[KEYS.status] === STATUS.pending) {
      self[KEYS.status] = STATUS.rejected
      self[KEYS.reason] = reason
      self[KEYS.rejectedCallbacks].forEach(callback => callback(reason))
    }
  }

  try {
    constructor(resolve, reject)
  } catch (err) {
    console.error(err)
  }
}

// 每次then之后，根据then返回的东西来判断具体执行逻辑
function resolver(result, resolve, reject) {
  // 根据
  if (result != null && (typeof result === 'object' || typeof result === 'function')) {
    let then = result.then
    if (typeof then === 'function') {
      then()
    } else {
      resolve(result)
    }
  } else {
    resolve(result)
  }
}

MyPromise.prototype.then = function (onfulfilled, onreject) {
  let self = this
  return promise = new MyPromise(function (resolve, reject) {
    setTimeout(function () {
      if (self[KEYS.status] === STATUS.pending) {
        console.log('still pending！')
        self[KEYS.fulfilledCallbacks].push(() => {
          let x = onfulfilled(self[KEYS.value])
          resolver(x, resolve, reject)
        })
        self[KEYS.rejectedCallbacks].push(() => {
          let x = onreject(self[KEYS.reason])
          resolver(x, resolve, reject)
        })
      }

      if (onfulfilled instanceof Function) {
        if (self[KEYS.status] === STATUS.fulfilled) {
          console.log('before onfulfilled callback')
          resolver(onfulfilled(self[KEYS.value]), resolve, reject)
        }
      }

      if (onreject instanceof Function) {
        if (self[KEYS.status] === STATUS.rejected) {
          console.log('before onreject callback')
          resolver(onreject(self[KEYS.reason]), resolve, reject)
        }
      }
    })
  })

}

优点:

let promise = new MyPromise(function (resolve, reject) {
  setTimeout(function () {
    console.log('async task')
    resolve('first call')
  }, 2000)
})
console.log('im runing first!')

promise.then(function (value) {
  console.log(value)
  return 'second call'
}).then(function (value) {
  console.log(value)
  console.log('third call')
})

// 输出

im runing first!
still pending！
still pending！
async task
resolve was called
first call
resolve was called
second call
third call
resolve was called
支持链式调用，支持异步

总结： 虽然第二种方法解决了链式调用问题，但离真正Promise A+规范还有一定差距， ES6 Promise 实现了Promise A+规范，有能力的可以阅读ES6 Promise源码
