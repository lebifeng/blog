# 统一调度 && 事件中心

传统web项目(mvc)前端开发过程中难免会遇到多模块协同开发的情况, 一般的做法是一个模块一个js, 分别维护各模块的生命周期, 状态等, 带来的问题就是各模块的初始化顺序难以控制, 模块间的通信也不好设计, 研究一个统一调度中心看起来很有必要,下面提供一个简易方案:

# 调度中心: 统一管控所有注册的模块

(function () {
  function Center() {
    this.name = "center"
    // 安装模块的实例都存在这
    this.modules = []
  }

// 主函数初始化, 依次调用各模块初始化函数
  Center.prototype.init = function () {
    console.log('im center, start initializing')
    console.log(this.modules)
    this.modules.forEach(e => {
      e.init()
    })

    document.querySelector('#button1')
    .addEventListener('click', (event) => {
      this.modules.forEach(e => {
        e.eventWatcher({ from: this.name, event, element: "#button1", params: '123'})
      })
    })
  }

// 通用查询方法
  Center.prototype.query = function (who) {
    console.log(`${who} is querying`)
    return 'result'
  }

  /**
   * 模块安装方法
   * @param module
   */
  Center.prototype.install = function (module) {

    // 子模块绑定主模块原型方法
    module.prototype.query = this.query.bind(this)
    module.prototype.dispatch = this.dispatch.bind(this)
    this.modules.push(new module())

    // 模块排序
    this.modules.sort((m1, m2) => {
      return m1.order - m2.order
    })
  }

  /**
   * 事件分发函数
   * @param from 发送者 required
   * @param event 事件|undefined
   * @param element 元素|undefined
   * @param params 参数|undefined
   * @param to 接受者|undefined
   * @returns {boolean}
   */
  Center.prototype.dispatch = function (from, event, element, params, to) {
    console.log(`im center, got your message, ${to ? `sending to ${to}` : 'sending to everyone else'}`)
    let notifiers = []

    // 有to的话,把接收模块找出来,否则默认发送给其他所有模块
    if (to) {
      notifiers = this.modules.filter(e => e.name === to)
    } else {
      notifiers = this.modules.filter(e => e.name !== from)
    }

    if (notifiers.length === 0) {
      return false
    }
    // 发起通知
    notifiers.forEach(e => {
      e.eventWatcher({ from ,event, element, params})
    })
  }
// 注册到全局, 方便各子模块注册
  window.center = new Center()
})()

# 模块A, 模块B

(function () {
  function ModuleA() {
    this.name = 'moduleA'
    this.order = 2
  }

  ModuleA.prototype.init = function () {
    console.log('im moduleA, start initializing')
    this.query(this.name)
    setTimeout(() => {
      this.dispatch(this.name, 'refresh', 'moduleA', 'param')
    }, 3000)

    document.querySelector('#button2')
    .addEventListener('click', (event) => {
      this.dispatch('moduleA', event, '#button2', '你好啊', 'moduleB')
    })
  }

  ModuleA.prototype.refresh = function () {
    console.log('im moduleA, start refreshing')
  }

  // 各模块自定义的事件处理函数
  ModuleA.prototype.eventWatcher = function (message) {
    console.log(`${this.name} got message from ${message.from}, message is ${JSON.stringify(message)}`)
  }

  window.center.install(ModuleA)
})()

(function () {
  function ModuleB() {
    this.name = 'moduleB'
    this.order = 1
  }

  ModuleB.prototype.init = function () {
    console.log('im moduleB, start initializing')
  }

  ModuleB.prototype.refresh = function () {
    console.log('im moduleB, start refreshing')
  }

  // 各模块自定义的事件处理函数
  ModuleB.prototype.eventWatcher = function (message) {
    console.log(`${this.name} got message from ${message.from}, message is ${JSON.stringify(message)}`)
  }

  window.center.install(ModuleB)
})()
